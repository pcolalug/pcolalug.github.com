Here's my script to grab a fortune from thinkgeek:

Code:
#!/bin/bash
echo;echo;wget -q http://www.thinkgeek.com/brain/fortune.cgi ;
sed -n '/blockquote/,/\/blockquote/p' ~/fortune.cgi |sed s/blockquote/thinkgeek/|sed s/\&quot\;/\"/g|sed s/\&\#39\;/\'/g|sed s/\&acute\;/\'/g|sed s/\<BR\>//g|sed s/\&lt\;/\</g|sed s/\&gt\;/\>/g|sed s/\&amp\;/\&/g|sed s/\&\#146\;/\'/g;echo;echo;
rm ~/fortune.cgi
