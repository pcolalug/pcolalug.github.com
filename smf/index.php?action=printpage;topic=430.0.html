Post by: rugmonster on January 28, 2003, 07:06:29 pm
Here is a copy of my script. I just read through it again and made some tweaks. I want to sit down and rewrite it so it fits the typical sysVinit style, but this should do the job. It is geared toward someone with a dynamic IP. If you have a static IP, just put it in the configuration section.

Enjoy!

Code:
#!/bin/sh

# Netfilter IPTables Firewall
# v 1.2 2003-01-29
# Daniel Givens
# <dgivens4 AT cox DOT net>
#
# Use and modify as you see fit.
#
# Sections taken from script written by Rick Dicaire
# His script can be found at http://linux.ardynet.com/ipmasq/ipmasq.php3


########## USER CONFIGURATION SECTION ###########

# Location of IPTABLES
IPT="/sbin/iptables"

# Internal Network Information
ITNET="192.168.1.0/24"
ITIP="192.168.1.1"

# External Network Information
EXIP=`ifconfig eth0| grep inet| cut -f2 -d:| cut -f1 -d" "`
EXNET=$EXIP/24

######### END USER CONFIGURATION SECTION #########

# Flush default chains
$IPT -F INPUT
$IPT -F OUTPUT
$IPT -F FORWARD
$IPT -t nat -F POSTROUTING
# For further instances, flush chains
# created later in this script
$IPT -F PRIV_PORT
$IPT -F UNPRIV_PORT
$IPT -F NL_DENY
$IPT -F SPOOF_SOURCE
$IPT -F BANNED
$IPT -F LOG_ACCEPT



######### Our custom chains to deal with unwanted traffic

# Denied Privileged Ports (1-1023)

$IPT -N PRIV_PORT
$IPT -A PRIV_PORT -m state --state RELATED,ESTABLISHED -j ACCEPT
$IPT -A PRIV_PORT -m limit -j LOG --log-level 5 --log-prefix "LOG_PRIV_PORT: "
$IPT -A PRIV_PORT -m state --state NEW,INVALID -j DROP

# Denied Unprivileged Ports (1024+)

$IPT -N UNPRIV_PORT
$IPT -A UNPRIV_PORT -m state --state RELATED,ESTABLISHED -j ACCEPT
$IPT -A UNPRIV_PORT -m limit -j LOG --log-level 5 --log-prefix "LOG_UNPRIV_PORT: "
$IPT -A UNPRIV_PORT -m state --state NEW,INVALID -j DROP

# Denied (Blatantly Obvious) Spoof Source Traffic

$IPT -N SPOOF_SOURCE
$IPT -A SPOOF_SOURCE -m limit -j LOG --log-level 5 --log-prefix "LOG_SPOOF_SOURCE: "
$IPT -A SPOOF_SOURCE -j DROP

# Banned Network

$IPT -N BANNED
$IPT -A BANNED -m limit -j LOG --log-level 5 --log-prefix "LOG_BANNED_NET: "
$IPT -A BANNED -j DROP

# Logging Accept

$IPT -N LOG_ACCEPT
$IPT -A LOG_ACCEPT -m limit -j LOG --log-level 5 --log-prefix "LOG_ACCEPTED_TRAFFIC: "
$IPT -A LOG_ACCEPT -j ACCEPT

######### Now to start defining some rules
 
### Setup masquerading

$IPT -t nat -A POSTROUTING -o eth0 -j MASQUERADE
echo 1 > /proc/sys/net/ipv4/ip_forward


### Setup flood protection

# Port Scans
$IPT -A INPUT -i eth0 -p tcp --tcp-flags SYN,ACK,FIN,RST RST -m limit --limit 2/s -j ACCEPT

# Syn-flood
$IPT -A INPUT -i eth0 -p tcp --syn -m limit --limit 3/s -j ACCEPT

# Ping o' Death
$IPT -A INPUT -i eth0 -p icmp --icmp-type echo-request -m limit --limit 1/s -j ACCEPT

### Internal and Trusted Networks
# allow all traffic on internal network
$IPT -A INPUT -i eth1 -s $ITNET -j ACCEPT

# allow traffic to lo
$IPT -A INPUT -i lo -j ACCEPT



### Trusted Networks from the outside
# To add more use the following format:
# $IPT -A INPUT [-p (protocol)] [-i (!) interface] [-s (source address/net)] [--sport (port)] [-d (destination address/net)] [--dport (port)] -j ACCEPT

# DHCP Servers (if you can put in a source address)
#$IPT -A INPUT -p UDP --sport 67 --dport 68 -j ACCEPT


### Banned Networks from the outside
# To add more use the following format:
# $IPT -A INPUT [-p (protocol)] [-i (!) interface] [-s (source address/net)] [--sport (port)] [-d (destination address/net)] [--dport (port)] -j BANNED

# Obviously Spoofed IPs from the outside
$IPT -A INPUT -i ! eth1 -s 192.168.0.0/16 -j SPOOF_SOURCE
$IPT -A INPUT -i eth1 -s ! 192.168.0.0/16 -j SPOOF_SOURCE
#$IPT -A INPUT -s 10.0.0.0/8 -j SPOOF_SOURCE
$IPT -A INPUT -s 172.16.0.0/16 -j SPOOF_SOURCE
$IPT -A INPUT -i ! lo -s 127.0.0.0/8 -j SPOOF_SOURCE



### Services
# Uncomment any that are needed.
# To add more use the following format:
# $IPT -A INPUT [-p (protocol)] [-i (!) interface] [-s (source address/net)] [--sport (port)] [-d (destination address/net)] [--dport (port)] -j ACCEPT(LOG_ACCEPT)

# ICMP (okay, not a service, but a protocol)
#$IPT -A INPUT -p icmp -j ACCEPT

# ftpd
#$IPT -A INPUT -p tcp -i eth0 -s ! $ITNET --dport 20 -j ACCEPT
#$IPT -A INPUT -p tcp -i eth0 -s ! $ITNET --dport 21 -j ACCEPT

# sshd
#$IPT -A INPUT -p tcp -i eth0 -s ! $ITNET --dport 22 -j LOG_ACCEPT
#$IPT -A INPUT -p udp -i eth0 -s ! $ITNET --dport 22 -j LOG_ACCEPT

# smtp
#$IPT -A INPUT -p tcp -i eth0 -s ! $ITNET --dport 25 -j ACCEPT

# httpd
#$IPT -A INPUT -p tcp -i eth0 -s ! $ITNET --dport 80 -j ACCEPT

# pop3d
#$IPT -A INPUT -p tcp -i eth0 -s ! $ITNET --dport 110 -j ACCEPT

# identd
#$IPT -A INPUT -p tcp -i eth0 -s ! $ITNET --dport 113 -j ACCEPT

# imapd
#$IPT -A INPUT -p tcp -i eth0 -s ! $ITNET --dport 143 -j ACCEPT

# httpsd
#$IPT -A INPUT -p tcp -i eth0 -s ! $ITNET --dport 443 -j ACCEPT

### Broad DENY Sets
# We will log everything that hasn't been specifically allowed above.
# Limits have been placed on the logging as to hopefully avoid
# building up a huge log file.

# TCP
$IPT -A INPUT -p tcp -s ! $ITNET --dport 1:1023 -j PRIV_PORT
$IPT -A INPUT -p tcp -s ! $ITNET --dport 1024:65535 -j UNPRIV_PORT

# UDP
$IPT -A INPUT -p udp -s ! $ITNET --dport 1:1023 -j PRIV_PORT
$IPT -A INPUT -p udp -s ! $ITNET --dport 1024:65535 -j UNPRIV_PORT

# Drop all FORWARD
$IPT -A FORWARD -j drop
